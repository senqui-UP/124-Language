/say --- functions and recursion ---

/function #fibonacci(@n) {
  /execute if @n <= 1 run {
    /return @n
  }
/return #fibonacci(@n - 1) + #fibonacci(@n - 2)
}

/say First 15 Fibonacci numbers
/summon int @val 0
/execute for @i in range(15) run {
  /set @val = {#fibonacci(@i)}
  /say fib({@i}) = {@val}
}

/say --- natives: clock and toString ---
/summon double @now 0
/summon string @numStr 0
/set @now = clock()
/say clock seconds = {@now}
/set @numStr = toString (12345)
/say toString on 12345 -> {@numStr}
