/whisper A simple program demonstrating variables and scope

/summon String @greeting
/set @greeting = Hello

/summon String @name
/set @name = World

/say {@greeting}, {@name}!

/summon int @x (10)
/summon int @y (20)

/execute if true run {
    /summon int @x (5)  
    /say Inner x: {@x}
    /say Outer y: {@y}
}

/say Outer x: {@x}
/set @x = {@x + @y}
/say x + y = {@x}
